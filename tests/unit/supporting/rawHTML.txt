function toggleOpenInterest()
{
	if (document.getElementById)
	{
		var visibleRowCount = 0;
		var table = document.getElementById('optionstable');
		var rows = table.getElementsByTagName("tr");
		for (var i = 1, limit = rows.length; i < limit; ++i)
		{
			var visible = true;
			var row = rows[i];
			if (row.className.indexOf("showzeroopeninterest") != -1)
			{
				row.className = "hidezeroopeninterest";
				visible = false;
			}
			else if (row.className.indexOf("hidezeroopeninterest") != -1)
			{
				row.className = "showzeroopeninterest";
			}
			else
			{
				row.classname = "";
			}
			if (visible)
			{
				++visibleRowCount;
				if (visibleRowCount % 2 == 0)
				{
					row.className = row.className + " altrow";
				}
			}
		}
	}
}
</script>
